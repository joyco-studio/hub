## Installation

```bash
npx shadcn@latest add https://r.joyco.studio/matrix.json
```

## Usage

```tsx
import { Matrix, wave } from '@/components/ui/matrix'

<Matrix
  rows={7}
  cols={7}
  frames={wave}
  fps={24}
  loop
  autoplay
/>
```
